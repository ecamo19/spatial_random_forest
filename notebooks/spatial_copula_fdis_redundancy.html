library(psych)
library(quantreg)
library(ggplot2)
# Load data
source(file = "./scripts/data_cleaned.R")

# Load models
load("./data/rdata_files/model_spatial_random_forest_fdis.RData")
load("./data/rdata_files/model_spatial_random_forest_redundancy.RData")
# Get fitted values
fitted_fdis <- model_spatial_random_forest_fdis$predicted
fitted_redundancy <- model_spatial_random_forest_redundancy$predicted

data_marginals <- as.data.frame(cbind(fitted_fdis, fitted_redundancy))

1. Simple correlation

pairs.panels(data_marginals,
    method = "spearman",
    smooth = TRUE,
    lm = TRUE,
    hist.col = "#1853b1",
    ci = TRUE
)

2. Quantile Regression

quantile_fit <- rq(fitted_fdis ~ fitted_redundancy, data = data_marginals)
summary(quantile_fit)
