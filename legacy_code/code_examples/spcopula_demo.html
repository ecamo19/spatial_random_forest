# devtools::install_github("BenGraeler/spcopula")
library(spcopula)
library(VineCopula)
library(sp)
# Load data
data("meuse")

Prepare data

# Transform data to SpatialPointsDataFrame
coordinates(meuse) <- ~ x + y
# Exploratory plots
spplot(meuse, "zinc", col.regions = bpy.colors(5))

hist(meuse[["zinc"]],
    freq = F, n = 30, ylim = c(0, 0.0035),
    main = "Histogram of zinc", xlab = "zinc concentration")

## log Transform
mean_log <- mean(log(meuse[["zinc"]]))
sd_log <- sd(log(meuse[["zinc"]]))
# Get marginal distributions
pMar <- function(q) plnorm(q, mean_log, sd_log)
qMar <- function(p) qlnorm(p, mean_log, sd_log)
dMar <- function(x) dlnorm(x, mean_log, sd_log)

Copula families

families <- list(
    normalCopula(0.3),
    tCopula(0.3, df = 2.15),
    VC2copula::joeBiCopula(1.5),
    VC2copula::surGumbelCopula(1),
    VC2copula::surJoeBiCopula(1.5))

Firts spatial tree

Rank data

# rank() = Returns the sample ranks of the values in a vector.
meuse$rt_zinc <- rank(meuse$zinc) / (length(meuse) + 1)
hist(meuse$rt_zinc)

Create lag classes

# For non-elliptical copulas you may want to use either Spearmans Rho or
# Kendalls Tau).

bins <- calcBins(meuse, var = "rt_zinc", nbins = 10, cutoff = 800)

# Calculate parameters for Kendall's tau function
kendall_tau_params <- fitCorFun(bins, degree = 2)

Call:
lm(formula = lagCor ~ poly(meanDists, degree), data = bins)

                  0.2076                   -0.5827                    0.1626

Calculate the log-likelihood

# Calculates the log-likelihood for a set of provided copula families per lag
# class
log_lik_tau <- loglikByCopulasLags(bins, meuse, families, kendall_tau_params)
